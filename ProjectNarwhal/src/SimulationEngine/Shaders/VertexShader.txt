 #version 150 core

  in vec3 position;
  in vec2 textureCoords;

  out vec2 TextureCoords;

  uniform mat4 transformationMatrix;
  uniform mat4 projectionMatrix;

  void main()
  {
      gl_Position = projectionMatrix * transformationMatrix * vec4(position, 1.0);
      TextureCoords = textureCoords;
  }
